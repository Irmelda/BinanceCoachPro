"""
BinanceCoach Pro — Bot Telegram Principal
"""

import os
import logging
import anthropic
from dotenv import load_dotenv
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import (
    Application, CommandHandler, MessageHandler,
    CallbackQueryHandler, ContextTypes, filters
)

# Import des skills
from skills.link_check import check_link, format_response as format_link
from skills.token_analysis import analyze_token
from skills.market_movement import analyze_market_movement
from skills.x_feed import get_news_summary
from skills.rug_pull import detect_rug_pull
from skills.launchpool_alerts import get_opportunities
from skills.pnl_tracker import get_account_pnl, request_api_keys_message

load_dotenv()

# Configuration
TELEGRAM_TOKEN = os.getenv("TELEGRAM_TOKEN")
ANTHROPIC_API_KEY = os.getenv("ANTHROPIC_API_KEY")

logging.basicConfig(
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
    level=logging.INFO
)

client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

# Stockage temporaire des clés utilisateur (en mémoire)
user_binance_keys = {}

# ============================================
# Détection de langue
# ============================================

def detect_language(text: str) -> str:
    """Détecte la langue du message"""
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=10,
        messages=[{
            "role": "user",
            "content": f"Detect the language of this text and respond with ONLY the 2-letter ISO code (fr, en, es, ar, pt, zh, etc.): '{text}'"
        }]
    )
    return response.content[0].text.strip().lower()[:2]


# ============================================
# Menu principal
# ============================================

def get_main_menu(lang: str = "fr") -> tuple:
    """Retourne le texte et le clavier du menu principal"""
    
    texts = {
        "fr": "🤖 *BinanceCoach Pro*\n\nTon assistant crypto intelligent.\nChoisis une option :",
        "en": "🤖 *BinanceCoach Pro*\n\nYour smart crypto assistant.\nChoose an option:",
        "es": "🤖 *BinanceCoach Pro*\n\nTu asistente crypto inteligente.\nElige una opción:",
        "ar": "🤖 *BinanceCoach Pro*\n\nمساعدك الذكي للعملات المشفرة.\nاختر خياراً:",
    }

    text = texts.get(lang, texts["en"])

    keyboard = [
        [
            InlineKeyboardButton("🔗 Link Check", callback_data="link_check"),
            InlineKeyboardButton("🪙 Token Analysis", callback_data="token_analysis"),
        ],
        [
            InlineKeyboardButton("📊 Market Movement", callback_data="market"),
            InlineKeyboardButton("🐦 Binance News", callback_data="news"),
        ],
        [
            InlineKeyboardButton("🛡️ Rug Pull Detector", callback_data="rugpull"),
            InlineKeyboardButton("📅 Launchpool", callback_data="launchpool"),
        ],
        [
            InlineKeyboardButton("💰 Mon P&L", callback_data="pnl"),
        ],
    ]

    return text, InlineKeyboardMarkup(keyboard)


# ============================================
# Handlers
# ============================================

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Commande /start"""
    lang = "fr"
    text, keyboard = get_main_menu(lang)
    await update.message.reply_text(text, reply_markup=keyboard, parse_mode="Markdown")


async def menu(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Commande /menu"""
    await start(update, context)


async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Gère les clics sur les boutons du menu"""
    query = update.callback_query
    await query.answer()
    
    user_id = query.from_user.id
    data = query.data

    # Stocker l'état de l'utilisateur
    context.user_data["waiting_for"] = data

    prompts = {
        "link_check": {
            "fr": "🔗 *Link Check*\n\nEnvoie-moi le lien à vérifier :",
            "en": "🔗 *Link Check*\n\nSend me the link to verify:",
        },
        "token_analysis": {
            "fr": "🪙 *Token Analysis*\n\nEnvoie-moi le symbole du token (ex: BTC, ETH, BNB) :",
            "en": "🪙 *Token Analysis*\n\nSend me the token symbol (e.g. BTC, ETH, BNB):",
        },
        "market": {
            "fr": "📊 *Market Movement*\n\nQue veux-tu savoir sur le marché ? (ou envoie 'analyse' pour un résumé général)",
            "en": "📊 *Market Movement*\n\nWhat do you want to know about the market?",
        },
        "rugpull": {
            "fr": "🛡️ *Rug Pull Detector*\n\nEnvoie l'adresse du contrat (et la chaîne si ce n'est pas BSC) :\nEx: `0x123...abc bsc`",
            "en": "🛡️ *Rug Pull Detector*\n\nSend the contract address (and chain if not BSC):\nEx: `0x123...abc eth`",
        },
        "pnl": {
            "fr": request_api_keys_message("fr"),
            "en": request_api_keys_message("en"),
        },
    }

    # Actions immédiates (pas besoin d'input)
    if data == "news":
        await query.edit_message_text("⏳ Récupération des dernières news Binance...")
        result = get_news_summary("fr")
        await query.edit_message_text(result, parse_mode="Markdown")
        return

    if data == "launchpool":
        await query.edit_message_text("⏳ Recherche des opportunités Launchpool...")
        result = get_opportunities("fr")
        await query.edit_message_text(result, parse_mode="Markdown")
        return

    # Actions nécessitant un input
    prompt = prompts.get(data, {}).get("fr", "Envoie ta question :")
    await query.edit_message_text(prompt, parse_mode="Markdown")


async def message_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Gère les messages texte des utilisateurs"""
    text = update.message.text
    user_id = update.message.from_user.id
    waiting_for = context.user_data.get("waiting_for")

    # Détecter la langue
    lang = detect_language(text)

    await update.message.reply_text("⏳ Analyse en cours...")

    try:
        # Router vers le bon skill
        if waiting_for == "link_check":
            result = check_link(text)
            response = format_link(result)
            context.user_data["waiting_for"] = None

        elif waiting_for == "token_analysis":
            response = analyze_token(text.strip())
            context.user_data["waiting_for"] = None

        elif waiting_for == "market":
            response = analyze_market_movement(text, lang)
            context.user_data["waiting_for"] = None

        elif waiting_for == "rugpull":
            parts = text.strip().split()
            contract = parts[0]
            chain = parts[1] if len(parts) > 1 else "bsc"
            response = detect_rug_pull(contract, chain)
            context.user_data["waiting_for"] = None

        elif waiting_for == "pnl" or text.startswith("/pnl"):
            # Format: /pnl API_KEY SECRET_KEY
            parts = text.replace("/pnl", "").strip().split()
            if len(parts) >= 2:
                api_key, secret_key = parts[0], parts[1]
                user_binance_keys[user_id] = {"api_key": api_key, "secret_key": secret_key}
                response = get_account_pnl(api_key, secret_key)
            else:
                response = request_api_keys_message(lang)
            context.user_data["waiting_for"] = None

        else:
            # Message libre → Claude répond en mode général
            with open("system_prompt.md", "r") as f:
                system = f.read()

            ai_response = client.messages.create(
                model="claude-sonnet-4-20250514",
                max_tokens=1000,
                system=system,
                messages=[{"role": "user", "content": text}]
            )
            response = ai_response.content[0].text

        await update.message.reply_text(response, parse_mode="Markdown")

    except Exception as e:
        await update.message.reply_text(f"❌ Erreur : {str(e)}\nRéessaie ou tape /menu")


# ============================================
# Lancement du bot
# ============================================

def main():
    app = Application.builder().token(TELEGRAM_TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("menu", menu))
    app.add_handler(CallbackQueryHandler(button_handler))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, message_handler))

    print("🤖 BinanceCoach Pro est en ligne !")
    app.run_polling()


if __name__ == "__main__":
    main()
